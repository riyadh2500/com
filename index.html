Quick Start
Rialo · Event-Driven Contract 
// Rialo native web call — no oracle needed
use rialo::prelude::*;

#[rialo::contract]
pub fn on_price_feed(ctx: Context, feed: PriceFeed) {
    let price = feed.get("BTC/USD");

    if price > 70_000 {
        ctx.emit(Event::PriceAlert {
            asset: "BTC",
            price,
            threshold: 70_000,
        });
    }
}

// Native HTTP call — built into the runtime
#[rialo::http_trigger]
pub async fn fetch_market_data() -> Result<MarketData> {
    let data = http::get("https://api.rialo.io/v1/markets")
        .await?;
    Ok(data.json::<MarketData>()?)
}

🗄️ Shelby — Cavalier Setup

The Cavalier Setup is the recommended way to onboard as a Shelby storage provider. Unpack the provider bundle, install the required dependencies, and register your node on the Aptos chain.

1
Download the Provider Bundle
Fetch the latest Shelby provider bundle from the official registry. This contains the node binary, config templates, and the Cavalier bootstrap script.
bash
# Download the Shelby Cavalier bundle
curl -fsSL https://releases.shelby.xyz/cavalier/latest.tar.gz \
  -o shelby-cavalier.tar.gz

# Verify the checksum
sha256sum -c shelby-cavalier.tar.gz.sha256
2
Unpack the Bundle
Extract the archive into your preferred installation directory. The bundle includes the shelby-node binary, Cavalier CLI, and default configuration files.
bash
# Create the install directory
mkdir -p ~/shelby-cavalier && cd ~/shelby-cavalier

# Unpack the bundle
tar -xzf ~/shelby-cavalier.tar.gz --strip-components=1

# Confirm contents
ls -la
# shelby-node  cavalier  config/  scripts/  README.md
3
Install Dependencies
Run the Cavalier dependency installer. This sets up the Aptos CLI, Shelby SDK, required Rust toolchain, and configures your system paths automatically.
bash
# Run the Cavalier dependency installer
./cavalier setup --install-deps

# This installs:
#   ✓ Rust toolchain (stable)
#   ✓ Aptos CLI v3.x
#   ✓ @shelby-protocol/sdk
#   ✓ shelby-node binary

# Verify everything is ready
./cavalier doctor
4
Configure Storage
Edit the provider config to set your storage capacity, fiber endpoint, and Aptos wallet. Cavalier uses this to register your node and begin earning storage rewards.
config/provider.toml
# Shelby Cavalier — provider.toml

[storage]
capacity_gb  = 500
data_dir     = "/data/shelby"
erasure_code = "reed-solomon-8-4"

[network]
fiber_endpoint = "fiber.shelby.xyz:9000"
public_ip      = "auto"

[aptos]
wallet  = "0xYOUR_WALLET_ADDRESS"
network = "mainnet"
5
Register & Start Node
Register your node on-chain and start the Shelby storage daemon. Once registered, your node begins serving blobs and earning micropayment rewards.
bash
# Register node on Aptos mainnet
./cavalier register --config config/provider.toml

# Start the storage daemon
./shelby-node start --config config/provider.toml

# Monitor node status
./cavalier status

# 🟢 Node registered · Serving 500GB · Earning APT

Install & Initialise
1inch — npm install + quote 
# Install
npm install @1inch/swap-sdk ethers

// 1inch_swap.ts
import { SDK, NetworkEnum } from '@1inch/swap-sdk';

const sdk = new SDK({
  url: 'https://api.1inch.dev/swap/v6.0',
  authKey: 'YOUR_1INCH_API_KEY',
});

// Get best quote: USDC → WETH on Ethereum
const quote = await sdk.swap.getQuote({
  src:    '0xA0b86991c6218b36c1d19D4a2e9Eb0cE3606eB48', // USDC
  dst:    '0xC02aaA39b223FE8D0A0e5C4F27eAD9083C756Cc2', // WETH
  amount: '1000000000', // 1000 USDC (6 decimals)
  from:   '0xYOUR_WALLET',
  slippage: 0.5,
  chainId: NetworkEnum.ETHEREUM,
});

console.log('Best route:', quote.toAmount, quote.protocols);

npm install @0x/swap-ts-sdk viem

// zerox_swap.ts
import { SwapApiClient } from '@0x/swap-ts-sdk';

const client = new SwapApiClient({
  apiKey: 'YOUR_0X_API_KEY',
  chainId: 1, // Ethereum mainnet
});

// Price quote (indicative)
const price = await client.getPrice({
  sellToken:   '0xA0b86991c6218b36c1d19D4a2e9Eb0cE3606eB48',
  buyToken:    '0xC02aaA39b223FE8D0A0e5C4F27eAD9083C756Cc2',
  sellAmount:  '1000000000',
});

// Firm quote with Permit2 calldata
const quote = await client.getQuote({
  ...price,
  takerAddress: '0xYOUR_WALLET',
});
console.log(quote.transaction.to, quote.transaction.data);

# Install Uniswap v4 SDK + dependencies
npm install @uniswap/v4-sdk @uniswap/sdk-core @uniswap/universal-router-sdk ethers

// uniswap_v4_swap.ts
import { Token, CurrencyAmount, TradeType, Percent } from '@uniswap/sdk-core';
import { Pool, Route, Trade }                         from '@uniswap/v4-sdk';
import { UniversalRouter }                            from '@uniswap/universal-router-sdk';
import { ethers }                                     from 'ethers';

const ETHEREUM = 1;
const USDC  = new Token(ETHEREUM, '0xA0b8...', 6,  'USDC');
const WETH  = new Token(ETHEREUM, '0xC02a...', 18, 'WETH');

// Fetch pool state from Uniswap v4 PoolManager
const pool = await Pool.fromChain(USDC, WETH, 3000, provider);

// Build swap route
const route = new Route([pool], USDC, WETH);
const trade = await Trade.exactIn(
  CurrencyAmount.fromRawAmount(USDC, '1000000000'),
  route,
);

// Encode via Universal Router
const { calldata, value } = UniversalRouter.swapCallParameters(trade, {
  slippageTolerance: new Percent(50, 10_000), // 0.5%
  deadline: Math.floor(Date.now() / 1000) + 60 * 20,
  recipient: '0xYOUR_WALLET',
});
